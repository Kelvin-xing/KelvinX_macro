%plot of cdf and pdf of lognormal
omega=0:0.05:5;
y=lognpdf(omega,-0.5,1); %mu=-0.5;sigma=1;
plot(omega,y);
y2=logncdf(omega,-0.5,1);
plot(omega,y2);

%draw sample from lognormal
omega=lognrnd(-0.5,1,1e6,1);%one million samples
mean(omega);%should be close to unit.
std(omega);
var(omega); %should be close to e-1, which is about 1.718
